lotvmodKhunt <- function(t, pop, pars) {
  with(as.list(c(pars, pop)), {

    hunt  <- if (prey > min_prey) rhunt * prey else 0

    dprey <- rprey * (1 - prey / K) * prey - alpha * prey * pred - hunt
    dpred <- eff * alpha * prey * pred - pmort * pred
    return(list(c(dprey, dpred)))
  })
}

# --- Model ---
lotvmodKhunt <- function(t, pop, pars) {
  with(as.list(c(pars, pop)), {
    hunt  <- if (prey > min_prey) rhunt * prey else 0
    dprey <- rprey * (1 - prey / K) * prey - alpha * prey * pred - hunt
    dpred <- eff * alpha * prey * pred - pmort * pred
    return(list(c(dprey, dpred)))
  })
}

# --- Sampling ---
n_runs  <- 500
set.seed(42)
days    <- seq(1, 500, by = 1)
currpop <- c(prey = 1000, pred = 50)

param_samples <- tibble(
  rprey    = runif(n_runs, 0.95 * 0.8, 0.95 * 1.2),
  alpha    = runif(n_runs, 0.01 * 0.8, 0.01 * 1.2),
  eff      = runif(n_runs, 0.6  * 0.8, 0.6  * 1.2),
  pmort    = runif(n_runs, 0.4  * 0.8, 0.4  * 1.2),
  K        = 2000, 
  rhunt    = 0.1,       # change this & re-run
  min_prey = 500
)
# --- Run simulations ---
run_model <- function(row) {
  p <- unlist(row)
  out <- tryCatch(
    as.data.frame(ode(func = lotvmodKhunt, y = currpop, times = days, parms = p)),
    error = function(e) NULL
  )
  if (is.null(out)) return(NA_real_)
  min(tail(out$prey, 250))
}

param_samples$min_prey_last250 <- map_dbl(1:n_runs, ~run_model(param_samples[.x, ]))
# Stability flag based on min prey threshold
param_samples <- param_samples %>%
  mutate(stable = min_prey_last250 > 100)

# --- Summary ---
